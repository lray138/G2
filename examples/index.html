Maybe Usage Example

use lray138\G2\Maybe;

$maybe = Maybe\Just::of(42);
$result = $maybe->map(fn($x) => $x * 2)->fold('Nothing here', fn($x) => "Value: $x");
echo $result; // Output: Value: 84
